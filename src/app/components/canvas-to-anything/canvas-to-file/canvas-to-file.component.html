    "$('.canvasToFile button').on('click', (event) => {
      event.preventDefault()

      const canvas = $('.canvas-wrapper canvas')[0]

      canvasToFile(canvas, 'image/png', (file) => {
        fileToImage(file, image => {
          $('.canvasToFile .wrapper').append(image)
        })

        $('.canvasToFile .code div').html(
          `${file}
          ${JSON.stringify({
            size: file.size,
            type: file.type
          }, null, 2)}`
        )
      })
    })"
    }}

    {{
    "//  Step: canvas --> file
    
    canvasToFile(canvas: HTMLCanvasElement, mime: string, cb: Function): void {
      canvas.toBlob((blob) => {
        cb(blob)
      }, mime, 1)
    }"

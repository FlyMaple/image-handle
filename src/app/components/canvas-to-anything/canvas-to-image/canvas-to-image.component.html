    {{
    "//  Step: canvas --> data url --> iamge
    
    canvasToImage(canvas: HTMLCanvasElement, mime: string, cb: Function): void {
      const dataUrl = canvas.toDataURL(mime, 1)
  
      this.dataUrlToImage(dataUrl, image => {
        cb(image)
      })
    }"
